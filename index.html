# Configuration LLM local
from crewai import Agent, Crew, Process, LLM
llm = LLM(
model="ollama/llama3.2:latest",
base_url="http://localhost:11434"
)
# Agent 1 — recherche
chercheur = Agent(
role="Chercheur Expert",
goal="Trouver des infos fascinantes",
llm=llm
)
# Agent 2 — rédaction
rédacteur = Agent(
role="Rédacteur Créatif",
llm=llm
)
# Orchestration séquentielle
crew = Crew(
agents=[chercheur, rédacteur],
process=Process.sequential
)
